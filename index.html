// ▼ さっきコピーしたGASのURLをここに貼り付ける ▼ const GAS_URL = "https://script.google.com/macros/s/AKfycbzgGjGsR7_lT48L5oQsWnly3BS0iIgL8jElHTmmGO7MrgVAIph4EmUT4EM_tS_hT2rpDg/exec"; let html5QrCode = null; let selectedMode = ""; // 二重読み込み防止用のメモ帳（数秒間だけ記憶する） let recentlyScanned = []; // 大学の座標設定 const UNIV_LAT = 34.743429; const UNIV_LNG = 137.388910; const ALLOWED_RADIUS = 500; function selectMode(mode) { selectedMode = mode; document.getElementById('setup').style.display = 'none'; document.getElementById('scan-container').style.display = 'block'; html5QrCode = new Html5Qrcode("reader"); html5QrCode.start({ facingMode: "environment" }, { fps: 10, qrbox: 250 }, onScanSuccess) .then(() => { document.getElementById('msg').innerText = "QRを連続でかざしてください"; document.getElementById('msg').style.color = "#333"; }) .catch((err) => { document.getElementById('msg').innerText = "カメラ起動エラー: カメラを許可してください"; document.getElementById('msg').style.color = "red"; }); } function resetPage() { if (html5QrCode && html5QrCode.isScanning) { html5QrCode.stop().then(finalizeReset).catch(finalizeReset); } else { finalizeReset(); } } function finalizeReset() { document.getElementById('scan-container').style.display = 'none'; document.getElementById('setup').style.display = 'block'; document.getElementById('msg').innerText = "カメラ準備中..."; document.getElementById('msg').style.color = "#333"; selectedMode = ""; recentlyScanned = []; // メモ帳もリセット } // ▼ ここが爆速スキャンの心臓部 ▼ function onScanSuccess(decodedText) { // 1. もし「ついさっき読んだばかりのQR」なら、無視して次へ（カメラは止まらない） if (recentlyScanned.includes(decodedText)) return; // 2. 新しいQRなら、メモ帳に記録して「5秒後」に消去する（5秒間は二重送信されない） recentlyScanned.push(decodedText); setTimeout(() => { recentlyScanned = recentlyScanned.filter(item => item !== decodedText); }, 5000); // 3. カメラを止めずに、画面の表示だけ一瞬で「受付！」に変える const msgDiv = document.getElementById('msg'); msgDiv.innerText = "✅ スキャン完了: " + decodedText; msgDiv.style.color = "#28a745"; // 0.5秒後に表示を元に戻す（次々と読み込んでいる感が出る） setTimeout(() => { if(msgDiv.innerText.includes(decodedText)) { msgDiv.innerText = "次のQRをかざしてください"; msgDiv.style.color = "#333"; } }, 500); // 4. 裏側でこっそりGASに送信する（待たなくてOK！） if (selectedMode === 'outside') { sendToGasInBackground(decodedText); } else { checkLocationInBackground(decodedText); } } function checkLocationInBackground(decodedText) { navigator.geolocation.getCurrentPosition( (pos) => { const dist = getDistance(pos.coords.latitude, pos.coords.longitude, UNIV_LAT, UNIV_LNG); if (dist <= ALLOWED_RADIUS) { sendToGasInBackground(decodedText); } else { // エラーの時だけ画面に警告を出す document.getElementById('msg').innerText = `❌ 範囲外です (${Math.round(dist)}m)`; document.getElementById('msg').style.color = "red"; } }, () => { document.getElementById('msg').innerText = "❌ 位置情報を許可してください"; document.getElementById('msg').style.color = "red"; }, { enableHighAccuracy: true, timeout: 5000 } ); } // ▼ 待たずに裏側で送信する処理 ▼ function sendToGasInBackground(decodedText) { fetch(GAS_URL, { method: 'POST', headers: { 'Content-Type': 'text/plain' }, body: JSON.stringify({ qrData: decodedText, mode: selectedMode }) }) .then(response => response.text()) .then(res => { // 裏側で成功したらコンソール（裏画面）に記録するだけ。画面は止めない。 console.log("GAS送信成功:", res); }) .catch(err => { console.error("GAS通信エラー:", err); }); } function getDistance(lat1, lng1, lat2, lng2) { const R = 6371000; const dLat = (lat2 - lat1) * Math.PI / 180; const dLng = (lng2 - lng1) * Math.PI / 180; const a = Math.sin(dLat/2)**2 + Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin(dLng/2)**2; return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); }
